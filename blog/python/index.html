fizzbuzz challenge

# if divisible by both 5 and 3   return fizz_buzz
# if divisible only by 5 return fizz
# if divisible only by 3 return buzz
# else return the passed no.

def fizz_buzz(input_no1):
    input_no = int(input_no1)
    # if input_no % 5 == 0 and input_no % 3 == 0:
    if (input_no % 5) == 0 and (input_no % 3 == 0):
        return  "fizz_buzz"
    elif input_no % 5 == 0:
        return  "fizz"
    elif input_no % 3 == 0:
        return  "buzz"
    else:
        return  input_no

while True:
    number = input("insert a number:  ")
    print(fizz_buzz(number))
